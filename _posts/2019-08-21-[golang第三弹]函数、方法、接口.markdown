----------------------------------     函数     ----------------------------------------

形式1

  func 函数名(参数1 类型， 参数2 类型 ....)(结果 类型){
      
  }
  实例  
  func add(a int, b int)(result int){
        result = a + b
        return    /*直接return即可 return后面可带返回的参数也可不带*/
  }

形式2

  func 函数名(参数1 类型， 参数2 类型 ....)(类型){
      
  }
  实例  
  func add(a int, b int)(int){
        result := a + b
        return  result
  }
